## üìä ANALYTICS DASHBOARD SETUP

### Google Analytics 4 Events

Add to `js/main.js`:

```javascript
// Track image section visibility
document.addEventListener('DOMContentLoaded', function() {
  
  // Hero image load time
  const heroImg = document.querySelector('.hero-background-image');
  if (heroImg) {
    heroImg.addEventListener('load', function() {
      gtag('event', 'hero_image_loaded', {
        'load_time': performance.now(),
        'image_size': this.naturalWidth + 'x' + this.naturalHeight
      });
    });
  }
  
  // Channel card clicks
  document.querySelectorAll('.channel-card').forEach(card => {
    card.addEventListener('click', function() {
      const channel = this.classList.contains('whatsapp') ? 'whatsapp' :
                     this.classList.contains('email') ? 'email' :
                     this.classList.contains('phone') ? 'phone' : 'calendar';
      
      gtag('event', 'contact_channel_click', {
        'channel_type': channel,
        'section': 'homepage_contact'
      });
    });
  });
  
  // Case study image clicks
  document.querySelectorAll('.case-study-card').forEach(card => {
    card.addEventListener('click', function() {
      const industry = this.querySelector('.industry-badge').textContent;
      
      gtag('event', 'case_study_click', {
        'industry': industry,
        'has_image': true
      });
    });
  });
  
  // Video thumbnail clicks
  document.querySelectorAll('.video-play-button').forEach(btn => {
    btn.addEventListener('click', function() {
      gtag('event', 'video_play', {
        'video_type': 'testimonial',
        'section': 'success_stories'
      });
    });
  });
  
});
```

## üö® COMMON ISSUES & SOLUTIONS

### Issue: Images Not Loading

**Symptoms:** Broken image icons, 404 errors in console

**Solution:**
```bash
# Verify file exists
ls images/hero-office-transformation.png

# Check file permissions
chmod 644 images/*.png

# Verify path in HTML
<img src="images/..." />  ‚úÖ Correct
<img src="./images/..." /> ‚úÖ Also works
<img src="/images/..." />  ‚ùå Wrong (absolute path)
```

---

### Issue: Slow Page Load

**Symptoms:** Google PageSpeed <90, LCP >2.5s

**Solution:**
```bash
# Run optimization script
bash optimize-images.sh

# Check file sizes
ls -lh images/ | grep -E "(hero|case-study)"

# Target: Hero <500KB, Others <300KB

# If still large, manually optimize
pngquant --quality=60-75 images/large-image.png --force
```

---

### Issue: Layout Breaks on Mobile

**Symptoms:** Images overflow, text overlaps

**Solution:**
```css
/* Add to image-enhanced-sections.css */
@media (max-width: 768px) {
  .hero-background-image,
  .channels-image,
  .credentials-image {
    max-width: 100% !important;
    height: auto !important;
  }
  
  .hero-content {
    padding: 60px 16px !important;
  }
}
```
